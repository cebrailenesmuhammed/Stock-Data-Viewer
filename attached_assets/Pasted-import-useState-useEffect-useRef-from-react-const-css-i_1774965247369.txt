function decisionStyle(d = "") {
  const u = d.toUpperCase();
  if (u.includes("GÜÇLÜ AL"))  return { cls: "b-green", label: "🔥 GÜÇLÜ AL" };
  if (u.includes("AL"))        return { cls: "b-green", label: "✅ AL" };
  if (u.includes("GÜÇLÜ SAT")) return { cls: "b-red",   label: "⛔ GÜÇLÜ SAT" };
  if (u.includes("SAT"))       return { cls: "b-red",   label: "📉 SAT" };
  return { cls: "b-amber", label: "⏸️ BEKLE" };
}

async function callClaude(symbol) {
  const res = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      model: "claude-sonnet-4-20250514",
      max_tokens: 1000,
      messages: [{
        role: "user",
        content: `Sen bir fintech AI analiz asistanısın. ${symbol} hissesi için kısa ve net Türkçe analiz yap.

SADECE şu JSON formatında yanıt ver (başka hiçbir şey yazma):
{
  "decision": "GÜÇLÜ AL veya AL veya BEKLE veya SAT veya GÜÇLÜ SAT",
  "confidence": "Yüksek veya Orta veya Düşük",
  "priceEst": "tahmini fiyat örn $185.40 veya ₺280.00",
  "changeEst": "günlük değişim örn +1.2% veya -0.8%",
  "reasoning": "2-3 cümle net Türkçe analiz açıklaması",
  "positives": ["güçlü yön 1", "güçlü yön 2", "güçlü yön 3"],
  "risks": ["risk 1", "risk 2"]
}`
      }]
    })
  });
  const data = await res.json();
  const raw = data.content[0].text.replace(/```json|```/g, "").trim();
  return JSON.parse(raw);
}
